server {
    listen 8080 default_server;
    listen [::]:8080 default_server;
    server_name los-rezagados.ifigueroap.cl www.los-rezagados.ifigueroap.cl;

    location ^~ /.well-known/acme-challenge/ {
        alias /var/www/certbot/.well-known/acme-challenge/;
        try_files $uri =404;
    }

    location /admin/ {
        proxy_pass http://adminer:8080/;
        
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location / {
        add_header Content-Type text/plain;
        return 200 "Hello World from Port 8080 (HTTP)";
    }
}

server {
    listen 8443 ssl default_server;
    listen [::]:8443 ssl default_server;
    server_name los-rezagados.ifigueroap.cl www.los-rezagados.ifigueroap.cl;

    ssl_certificate /etc/letsencrypt/live/los-rezagados.ifigueroap.cl/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/los-rezagados.ifigueroap.cl/privkey.pem;
    ssl_trusted_certificate /etc/letsencrypt/live/los-rezagados.ifigueroap.cl/chain.pem;

    location /admin/ {
        proxy_pass http://adminer:8080/;

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location / {
        add_header Content-Type text/plain;
        return 200 "Hello World from Port 8443 (HTTPS)";
    }
}
