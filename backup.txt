services:
  proxy:
    image: nginx:alpine
    container_name: g10-proxy
    restart: always
    ports:
      - "${HOST_HTTP_PORT}:80"
    volumes:
      - ./proxy/conf:/etc/nginx/conf.d
    depends_on:
      - adminer
    networks:
      - g10-network

  mariadb:
    image: mariadb:11.8
    container_name: g10-mariadb
    restart: always
    environment:
      - MARIADB_ROOT_PASSWORD=${DB_ROOT_PASSWORD}
    volumes:
      - ./mariadb/data:/var/lib/mysql
    networks:
      - g10-network

  adminer:
    image: adminer
    container_name: g10-adminer
    restart: always
    depends_on:
      - mariadb
    networks:
      - g10-network

networks:
  g10-network:
    driver: bridge
    name: g10-network
